-- Function to send chat message (with cooldown protection)
local lastChatTime = 0
local function sendChatMessage(message)
    local currentTime = tick()
    -- Only send if at least 2 seconds have passed since last message
    if currentTime - lastChatTime >= 2 then
        -- Try different chat methods
        local success = pcall(function()
            -- Method 1: TextChatService (newer method)
            if game:GetService("TextChatService").TextChannels:FindFirstChild("RBXGeneral") then
                game:GetService("TextChatService").TextChannels.RBXGeneral:SendAsync(message)
            -- Method 2: Legacy chat system
            elseif ReplicatedStorage:FindFirstChild("DefaultChatSystemChatEvents") then
                ReplicatedStorage.DefaultChatSystemChatEvents.SayMessageRequest:FireServer(message, "All")
            else
                -- Method 3: Player Chat Method (Fallback)
                player:Chat(message)
            end
        end)
        
        if success then
            lastChatTime = currentTime
            return true
        else
            print("Chat failed - Message not sent: " .. message)
            return false
        end
    else
        print("Chat cooldown - Message not sent: " .. message)
        return false
    end
end

-- Function to get all available items from ReplicatedStorage.Items
local function getAllItems()
    -- Cache items for 30 seconds to avoid repeated scanning
    if allItemsCache and tick() - lastItemsCacheTime < 30 then
        return allItemsCache
    end
    
    local items = {}
    local itemsFolder = ReplicatedStorage:FindFirstChild("Items")
    
    if not itemsFolder then
        warn("Items folder not found in ReplicatedStorage!")
        return items
    end
    
    -- Get all tool names from the Items folder
    for _, tool in ipairs(itemsFolder:GetChildren()) do
        if tool:IsA("Tool") then
            table.insert(items, tool.Name)
        end
    end
    
    -- Sort alphabetically
    table.sort(items)
    
    -- Update cache
    allItemsCache = items
    lastItemsCacheTime = tick()
    
    return items
end

-- Function to find best matching username
local function findMatchingUsername(input, excludeSelf)
    if not input or input == "" then
        return nil, "Please provide a username"
    end
    
    input = input:lower()
    local matches = {}
    
    -- Get all players
    for _, playerObj in pairs(Players:GetPlayers()) do
        local playerName = playerObj.Name
        local playerNameLower = playerName:lower()
        
        -- Skip self if requested
        if excludeSelf and playerObj == player then
            continue
        end
        
        -- Check if input matches start of username
        if playerNameLower:sub(1, #input) == input then
            table.insert(matches, playerName)
        end
    end
    
    -- Sort matches alphabetically
    table.sort(matches)
    
    if #matches == 0 then
        return nil, "No users found matching '" .. input .. "'"
    elseif #matches == 1 then
        return matches[1], nil
    else
        -- Multiple matches found
        local matchList = table.concat(matches, ", ")
        return nil, "Multiple users found: " .. matchList .. ". Please be more specific."
    end
end

-- Function to find best matching player object
local function findMatchingPlayer(input, excludeSelf)
    local username, error = findMatchingUsername(input, excludeSelf)
    if error then
        return nil, error
    end
    
    local playerObj = Players:FindFirstChild(username)
    if playerObj then
        return playerObj, nil
    else
        return nil, "Player '" .. username .. "' not found in server"
    end
end

-- Function to find best matching item from all available items (for !bd)
local function findMatchingAnyItem(input)
    if not input or input == "" then
        return nil, "Please provide an item name"
    end
    
    input = input:lower()
    local exactMatches = {}
    local partialMatches = {}
    
    -- Get all available items
    local allItems = getAllItems()
    
    for _, itemName in ipairs(allItems) do
        local itemNameLower = itemName:lower()
        
        -- Check for exact match (case insensitive)
        if itemNameLower == input then
            table.insert(exactMatches, itemName)
        -- Check if input matches start of item name
        elseif itemNameLower:sub(1, #input) == input then
            table.insert(partialMatches, itemName)
        end
    end
    
    -- Sort matches alphabetically
    table.sort(exactMatches)
    table.sort(partialMatches)
    
    -- Prioritize exact matches
    if #exactMatches == 1 then
        local matchedItem = exactMatches[1]
        -- Block Phoenix purchases
        if matchedItem:lower() == "phoenix" then
            return nil, "NO - Cannot purchase Phoenix!"
        end
        return matchedItem, nil
    elseif #exactMatches > 1 then
        -- Check if any exact matches are Phoenix
        for _, itemName in ipairs(exactMatches) do
            if itemName:lower() == "phoenix" then
                return nil, "NO - Cannot purchase Phoenix!"
            end
        end
        local matchList = table.concat(exactMatches, ", ")
        return nil, "Multiple exact matches found: " .. matchList .. ". Please be more specific."
    end
    
    -- If no exact matches, use partial matches
    if #partialMatches == 0 then
        return nil, "No items found matching '" .. input .. "'"
    elseif #partialMatches == 1 then
        local matchedItem = partialMatches[1]
        -- Block Phoenix purchases
        if matchedItem:lower() == "phoenix" then
            return nil, "NO - Cannot purchase Phoenix!"
        end
        return matchedItem, nil
    else
        -- Check if any partial matches are Phoenix
        for _, itemName in ipairs(partialMatches) do
            if itemName:lower() == "phoenix" then
                return nil, "NO - Cannot purchase Phoenix!"
            end
        end
        -- Multiple partial matches found
        local matchList = table.concat(partialMatches, ", ")
        return nil, "Multiple items found: " .. matchList .. ". Please be more specific."
    end
end

-- Function to parse buy command (basic version for !buy)
local function parseBuyCommand(message)
    -- Remove the command prefix and trim spaces
    local cleanMessage = message:gsub("^!%a+%s*", ""):gsub("%s+", " ")
    
    -- Look for pattern: number-hyphen-text
    local amount, itemName = cleanMessage:match("^(%d+)%-(.+)$")
    
    if amount and itemName then
        -- Clean up the item name (remove extra spaces)
        itemName = itemName:gsub("^%s+", ""):gsub("%s+$", "")
        return tonumber(amount), itemName
    end
    
    return nil, nil
end

-- Function to parse buy command with autocomplete for ALL items (for !bd)
local function parseBuyCommandWithAutocomplete(message)
    -- Remove the command prefix and trim spaces
    local cleanMessage = message:gsub("^!%a+%s*", ""):gsub("%s+", " ")
    
    -- Look for pattern: number-hyphen-text
    local amount, itemInput = cleanMessage:match("^(%d+)%-(.+)$")
    
    if amount and itemInput then
        -- Clean up the item input (remove extra spaces)
        itemInput = itemInput:gsub("^%s+", ""):gsub("%s+$", "")
        
        -- Try to find matching item from ALL items
        local matchedItem, error = findMatchingAnyItem(itemInput)
        if error then
            return nil, nil, error
        end
        
        return tonumber(amount), matchedItem, nil
    end
    
    return nil, nil, "Invalid format. Use: amount-itemname"
end

-- Function to teleport to specific player
local function teleportToPlayer(targetPlayer)
    local targetCharacter = targetPlayer.Character
    local playerCharacter = player.Character
    
    if targetCharacter and playerCharacter then
        local targetHRP = targetCharacter:FindFirstChild("HumanoidRootPart")
        local playerHRP = playerCharacter:FindFirstChild("HumanoidRootPart")

-- Function to create drop platform relative to specific player
local function createDropPlatform(targetPlayer)
    -- Clean up existing platform
    if dropPlatform then
        dropPlatform:Destroy()
        dropPlatform = nil
    end
    
    -- Create new platform
    dropPlatform = Instance.new("Part")
    dropPlatform.Name = "DropPlatform"
    dropPlatform.Size = Vector3.new(3, 1, 3)
    dropPlatform.Anchored = true
    dropPlatform.CanCollide = true
    dropPlatform.Material = Enum.Material.Plastic
    dropPlatform.BrickColor = BrickColor.new("Bright green")
    dropPlatform.Parent = workspace
    
    -- Position platform at fixed height relative to target player
    local targetCharacter = targetPlayer.Character
    if targetCharacter then
        local targetHRP = targetCharacter:FindFirstChild("HumanoidRootPart")
        if targetHRP then
            -- Always create platform at fixed 25 studs above target player's position
            dropPlatform.Position = targetHRP.Position + Vector3.new(0, 25, 0)
        end
    end
    
    return dropPlatform
end

-- Function to delete Cold Plunge objects

end

end

-- Function to auto-whitelist specific users
local function autoWhitelistUsers()
    local whitelistedCount = 0
    
    for _, targetUsername in ipairs(AUTO_WHITELIST_USERS) do
        local playerObj = Players:FindFirstChild(targetUsername)
        if playerObj and not whitelistedUsers[targetUsername] then
            -- Add to whitelist
            whitelistedUsers[targetUsername] = true
            whitelistedCount = whitelistedCount + 1
            print("Auto-whitelisted: " .. targetUsername)
            sendChatMessage("Auto-whitelisted: " .. targetUsername)
        end
    end
    
    if whitelistedCount > 0 then
        print("Auto-whitelist: Added " .. whitelistedCount .. " users to whitelist")
    end
end

-- Function to start auto-whitelist
local function startAutoWhitelist()
    if autoWhitelistEnabled then return end
    
    autoWhitelistEnabled = true
    print("Auto-whitelist system started!")
    sendChatMessage("Auto-whitelist system started!")
    
    -- Run immediately once
    autoWhitelistUsers()
    
    -- Then run every 5 seconds
    autoWhitelistConnection = game:GetService("RunService").Heartbeat:Connect(function()
        if not autoWhitelistEnabled then return end
        autoWhitelistUsers()
        task.wait(5) -- Wait 5 seconds between checks
    end)
end

-- Function to stop auto-whitelist
local function stopAutoWhitelist()
    if not autoWhitelistEnabled then return end
    
    autoWhitelistEnabled = false
    if autoWhitelistConnection then
        autoWhitelistConnection:Disconnect()
        autoWhitelistConnection = nil
    end
    print("Auto-whitelist system stopped!")
    sendChatMessage("Auto-whitelist system stopped!")
end

-- Function to get all required items for a recipe (including nested recipes)
local function getRequiredItems(itemName, quantity)
    local requiredItems = {}
    
    local function addItems(recipe, mult)
        for _, item in pairs(recipe) do
            if CRAFTING_RECIPES[item] then
                -- This is a craftable item, get its ingredients
                addItems(CRAFTING_RECIPES[item], mult)
            else
                -- This is a base item
                requiredItems[item] = (requiredItems[item] or 0) + mult
            end
        end
    end
    
    if CRAFTING_RECIPES[itemName] then
        addItems(CRAFTING_RECIPES[itemName], quantity)
    end
    
    return requiredItems
end

-- Function to start buy/drop loop with Dropped item detection
local function startBuyDrop(amount, itemName, commandUser)
    if buyDropEnabled then 
        print("Buy/Drop already running! Use !sbd to stop first.")
        sendChatMessage("Buy/Drop already running! Use !sbd to stop first.")
        return 
    end
    
    buyDropEnabled = true
    currentPhase = "buying"
    local count = 0
    local completed = false
    local droppedConnection = nil
    
    -- Store the command user for teleporting back
    currentCommandUser = commandUser
    
    print("Starting Buy/Drop: " .. amount .. "x " .. itemName)
    sendChatMessage("Starting Buy/Drop: " .. amount .. "x " .. itemName)
    
    -- Create drop platform relative to command user
    createDropPlatform(commandUser)
    
    -- Function to handle dropped item detection
    local function onDroppedItemAdded(descendant)
        if not buyDropEnabled then return end
        
        -- Check if this is a dropped item in the Dropped folder
        local droppedFolder = workspace:FindFirstChild("Dropped")
        if not droppedFolder then return end
        
        -- Check if the descendant is inside the Dropped folder
        local isInDropped = false
        local parent = descendant.Parent
        while parent do
            if parent == droppedFolder then
                isInDropped = true
                break
            end
            parent = parent.Parent
        end
        
        if isInDropped and descendant:IsA("Part") then
            -- Check if this dropped item has an Owner and matches our item
            local owner = descendant:FindFirstChild("Owner")
            if owner and owner.Value == player and descendant.Name == itemName then
                print("Detected dropped item: " .. descendant:GetFullName())
                
                -- Delete the dropped item
                descendant:Destroy()
                print("Deleted dropped item: " .. itemName)
                
                -- Increment count and check if we're done
                count = count + 1
                print("Dropped and deleted: " .. itemName .. " (" .. count .. "/" .. amount .. ")")
                
                -- Send progress messages at 50% and 100%
                local progress = count / amount
                if progress == 0.5 then
                    sendChatMessage("Buy/Drop 50% complete!")
                elseif progress == 1.0 then
                    sendChatMessage("Buy/Drop 100% complete!")
                end
                
                -- Check if we're done
                if count >= amount then
                    sendChatMessage("Buy/Drop completed! " .. amount .. "x " .. itemName)
                    print("Buy/Drop completed!")
                    -- Teleport back to the user who sent the command
                    if currentCommandUser then
                        teleportToPlayer(currentCommandUser)
                        sendChatMessage("Returned to " .. currentCommandUser.Name)
                    end
                    stopBuyDrop()
                else
                    -- Continue with next cycle
                    currentPhase = "buying"
                end
            end
        end
    end
    
    -- Set up the dropped item listener
    droppedConnection = workspace.DescendantAdded:Connect(onDroppedItemAdded)
    
    buyDropConnection = game:GetService("RunService").Heartbeat:Connect(function()
        if not buyDropEnabled then
            if droppedConnection then
                droppedConnection:Disconnect()
            end
            cleanupDropPlatform()
            return 
        end
        
        if completed then return end
        
        if currentPhase == "buying" then
            if count < amount then
                -- Buy phase
                if buyItem(itemName) then
                    print("Bought: " .. itemName .. " (" .. (count + 1) .. "/" .. amount .. ")")
                end
                currentPhase = "dropping"
            end
        elseif currentPhase == "dropping" then
            -- Drop phase - teleport to platform first
            if teleportToDropPlatform() then
                -- Drop phase
                if dropItem(itemName) then
                    print("Dropped: " .. itemName)
                end
                currentPhase = "waiting_for_delete"
            else
                warn("Failed to teleport to drop platform!")
                sendChatMessage("Buy/Drop failed - teleport error!")
                -- Teleport back to the user who sent the command even on error
                if currentCommandUser then
                    teleportToPlayer(currentCommandUser)
                    sendChatMessage("Returned to " .. currentCommandUser.Name)
                end
                stopBuyDrop()
            end
        end
        -- Waiting_for_delete phase is handled by the dropped item listener
    end)
end

-- Function to stop buy/drop immediately
local function stopBuyDrop()
    if not buyDropEnabled then return end
    
    buyDropEnabled = false
    currentPhase = "idle"
    
    if buyDropConnection then
        buyDropConnection:Disconnect()
        buyDropConnection = nil
    end
    
    -- Clean up platform
    cleanupDropPlatform()
    
    -- Teleport back to the command user when stopped
    if currentCommandUser then
        teleportToPlayer(currentCommandUser)
        sendChatMessage("Returned to " .. currentCommandUser.Name)
    end
    
    print("Buy/Drop stopped immediately!")
    sendChatMessage("Buy/Drop stopped!")
end


-- Function to start orbiting specific player
local function startOrbit(targetPlayer)
    if orbitEnabled then return end
    
    orbitEnabled = true
    print("Orbit started!")
    sendChatMessage("Starting orbit around " .. targetPlayer.Name)
    
    local angle = 0
    
    orbitConnection = game:GetService("RunService").Heartbeat:Connect(function()
        if not orbitEnabled then return end
        
        local targetCharacter = targetPlayer.Character
        local playerCharacter = player.Character
        
        if targetCharacter and playerCharacter then
            local targetHRP = targetCharacter:FindFirstChild("HumanoidRootPart")
            local playerHRP = playerCharacter:FindFirstChild("HumanoidRootPart")
            
            if targetHRP and playerHRP then
                -- Calculate orbit position
                angle = angle + orbitSpeed * 0.1
                if angle >= 360 then angle = 0 end
                
                local x = math.cos(angle) * orbitDistance
                local z = math.sin(angle) * orbitDistance
                
                -- Position player in orbit around target
                local orbitPosition = targetHRP.Position + Vector3.new(x, 3, z)
                playerHRP.CFrame = CFrame.new(orbitPosition, targetHRP.Position)
            else
                sendChatMessage("Orbit stopped - character parts missing!")
                stopOrbit()
            end
        else
            sendChatMessage("Orbit stopped - characters not found!")
            stopOrbit()
        end
    end)
end

end

-- Function to start autocraft process
local function startAutocraft(quantity, itemName, commandUser)
    if autocraftEnabled then 
        sendChatMessage("Autocraft already running!")
        return 
    end
    
    if not CRAFTING_RECIPES[itemName] then
        sendChatMessage("Unknown recipe: " .. itemName)
        return
    end
    
    autocraftEnabled = true
    currentCommandUser = commandUser
    
    sendChatMessage("Please wait! Bot is currently crafting " .. quantity .. "x " .. itemName)
    
    -- Create main platform for final crafting
    createDropPlatform(commandUser)
    
    -- Run the autocraft process once (not in a loop)
    coroutine.wrap(function()
        -- Get all required base items
        local requiredItems = getRequiredItems(itemName, quantity)
        
        -- Buy all required base items
        sendChatMessage("Buying required materials...")
        for itemName, itemQuantity in pairs(requiredItems) do
            if not autocraftEnabled then break end
            sendChatMessage("Buying " .. itemQuantity .. "x " .. itemName)
            for i = 1, itemQuantity do
                if not autocraftEnabled then break end
                buyItem(itemName)
                print("Bought: " .. itemName .. " (" .. i .. "/" .. itemQuantity .. ")")
                if i < itemQuantity then
                    task.wait(1)
                end
            end
            if not autocraftEnabled then break end
            task.wait(1)
        end
        if not autocraftEnabled then 
            cleanupDropPlatform()
            return 
        end
        
        -- Check if there are intermediate items to craft
        local hasIntermediateItems = false
        if CRAFTING_RECIPES[itemName] then
            for _, ingredient in pairs(CRAFTING_RECIPES[itemName]) do
                if CRAFTING_RECIPES[ingredient] then
                    hasIntermediateItems = true
                    break
                end
            end
        end
        
        -- Craft intermediate items on faraway platform (to prevent dropping)
        if hasIntermediateItems then
            sendChatMessage("Crafting intermediate items...")
            local farPlatform = createFarawayPlatform()
            
            local function craftIfNeeded(ingredientName, qty)
                if CRAFTING_RECIPES[ingredientName] then
                    sendChatMessage("Crafting " .. qty .. "x " .. ingredientName)
                    
                    -- Teleport to faraway platform for intermediate crafting
                    if teleportToFarawayPlatform(farPlatform) then
                        sendChatMessage("Moved to safe location for intermediate crafting")
                    end
                    
                    for i = 1, qty do
                        if not autocraftEnabled then break end
                        craftItem(CRAFTING_RECIPES[ingredientName])
                        print("Crafted: " .. ingredientName .. " (" .. i .. "/" .. qty .. ")")
                        
                        -- The craft remote automatically drops, but we're far away so items return to inventory
                        if i < qty then
                            task.wait(1)
                        end
                    end
                    
                    -- Return to main platform after intermediate crafting
                    if teleportToDropPlatform() then
                        sendChatMessage("Returned to main platform")
                    end
                    
                    task.wait(1)
                end
            end
            
            -- Craft all intermediate items needed for the target item
            if CRAFTING_RECIPES[itemName] then
                for _, ingredient in pairs(CRAFTING_RECIPES[itemName]) do
                    if not autocraftEnabled then break end
                    if CRAFTING_RECIPES[ingredient] then
                        craftIfNeeded(ingredient, quantity)
                    end
                end
            end
            
            -- Clean up far platform
            if farPlatform then
                farPlatform:Destroy()
            end
        else
            sendChatMessage("No intermediate items needed - proceeding to final craft")
        end
        
        if not autocraftEnabled then 
            cleanupDropPlatform()
            return 
        end
        
        -- Always teleport to main platform for final crafting
        if not hasIntermediateItems then
            -- If no intermediates, teleport to platform now for final crafting
            if teleportToDropPlatform() then
                sendChatMessage("Teleported to crafting platform")
            else
                sendChatMessage("Failed to teleport to platform!")
                stopAutocraft()
                return
            end
        end
        
        -- Craft the final item on main platform (this will auto-drop)
        sendChatMessage("Crafting final item: " .. itemName)
        for i = 1, quantity do
            if not autocraftEnabled then break end
            craftItem(CRAFTING_RECIPES[itemName])
            print("Crafted: " .. itemName .. " (" .. i .. "/" .. quantity .. ")")
            
            -- No need to equip/drop since craft remote auto-drops
            -- Just wait between crafts
            if i < quantity then
                task.wait(1)
            end
        end
        
        sendChatMessage("Autocraft completed! Crafted " .. quantity .. "x " .. itemName)
        
        -- Teleport back to command user
        if currentCommandUser then
            teleportToPlayer(currentCommandUser)
            sendChatMessage("Returned to " .. currentCommandUser.Name)
        end
        
        stopAutocraft()
    end)()
end

-- Function to stop autocraft
local function stopAutocraft()
    if not autocraftEnabled then return end
    
    autocraftEnabled = false
    if autocraftConnection then
        autocraftConnection:Disconnect()
        autocraftConnection = nil
    end
    
    -- Clean up platform

    -- Teleport back to command user when stopped
    if currentCommandUser then

-- Function to handle chat messages
local function onPlayerChatted(message, speaker)
    local command = message:lower()
    
    -- !wl command - Add user to whitelist (Bot Owner Only)
    if command:sub(1,3) == "!wl" and isBotOwner(speaker) then
        local username = message:sub(4):gsub("^%s+", ""):gsub("%s+$", "")
        if username ~= "" then
            local matchedUsername, error = findMatchingUsername(username, false)
            if error then
                sendChatMessage(error)
            else
                local success, result = addToWhitelist(matchedUsername)
                sendChatMessage(result)
                print("Whitelist add attempt: " .. matchedUsername .. " - " .. result)
            end
        else
            sendChatMessage("Usage: !wl username")
        end
        return
    
    -- !uwl command - Remove user from whitelist (Bot Owner Only)
    elseif command:sub(1,4) == "!uwl" and isBotOwner(speaker) then
        local username = message:sub(5):gsub("^%s+", ""):gsub("%s+$", "")
        if username ~= "" then
            local matchedUsername, error = findMatchingUsername(username, false)
            if error then
                sendChatMessage(error)
            else
                local success, result = removeFromWhitelist(matchedUsername)
                sendChatMessage(result)
                print("Whitelist remove attempt: " .. matchedUsername .. " - " .. result)
            end
        else
            sendChatMessage("Usage: !uwl username")
        end
        return
    
    -- !wlstatus command - Show whitelist status (Bot Owner Only)
    elseif command == "!wlstatus" and isBotOwner(speaker) then
        local status = getWhitelistStatus()
        sendChatMessage(status)
        return
    end
    
    -- Check if the speaker is whitelisted for all other commands
    if not isWhitelistedUser(speaker) then return end
    
    -- !help command - Show help
    if command == "!help" then
        showHelp()
    
    -- !b command - Bring player to the user who sent the command
    elseif command == "!b" then
        local success = teleportToPlayer(speaker)
        if success then
            sendChatMessage("Teleported to " .. speaker.Name)
        else
            sendChatMessage("Failed to teleport to " .. speaker.Name)
        end
    
    -- !a command - Fire jump remote once
    elseif command == "!a" then
        fireJumpRemote()
        print("Jump remote fired once!")
        sendChatMessage("Jump executed!")
    
    -- !toa command - Start auto jumping (every 0.7 seconds)
    elseif command == "!toa" then
        startAutoJump()
    
    -- !tsa command - Stop auto jumping
    elseif command == "!tsa" then
        stopAutoJump()
    
    -- !buy command - Buy items
    elseif command:sub(1,4) == "!buy" then
        local amount, itemName = parseBuyCommand(message)
        if amount and itemName then
            if amount > 0 then
                -- Stop any existing buy/drop first
                stopBuyDrop()
                task.wait(0.1)
                
                -- Start buying only
                print("Starting buy: " .. amount .. "x " .. itemName)
                sendChatMessage("Starting buy: " .. amount .. "x " .. itemName)
                for i = 1, amount do
                    if not buyDropEnabled then
                        buyItem(itemName)
                        print("Bought: " .. itemName .. " (" .. i .. "/" .. amount .. ")")
                        if i == math.floor(amount * 0.5) then
                            sendChatMessage("Buy 50% complete!")
                        elseif i == amount then
                            sendChatMessage("Buy 100% complete!")
                        end
                        if i < amount then
                            task.wait(1) -- 1 second delay between buys
                        end
                    else
                        break
                    end
                end
                print("Buy completed!")
            else
                warn("Invalid amount: " .. tostring(amount))
            end
        else
            warn("Invalid !buy format. Use: !buy amount-itemname")
            warn("Example: !buy 5-8Bit Pineapple")
            warn("Your input: " .. message)
        end
    
    -- !bd command - Buy and drop loop (uses ALL items from ReplicatedStorage.Items)
    elseif command:sub(1,3) == "!bd" then
        local amount, itemName, error = parseBuyCommandWithAutocomplete(message)
        if error then
            sendChatMessage(error)
        elseif amount and itemName then
            if amount > 0 then
                stopBuyDrop()
                task.wait(0.1)
                startBuyDrop(amount, itemName, speaker)
            else
                sendChatMessage("Invalid amount: " .. tostring(amount))
            end
        end
    
    -- !sbd command - Stop buy/drop
    elseif command == "!sbd" then
        stopBuyDrop()
    
    -- !ac command - Autocraft items (uses CRAFTABLE items only)
    elseif command:sub(1,3) == "!ac" then
        local amount, itemName, error = parseCraftCommandWithAutocomplete(message)
        if error then
            sendChatMessage(error)
        elseif amount and itemName then
            if amount > 0 then
                stopAutocraft()
                task.wait(0.1)
                startAutocraft(amount, itemName, speaker)
            else
                sendChatMessage("Invalid amount: " .. tostring(amount))
            end
        end
    
    -- !sac command - Stop autocraft
    elseif command == "!sac" then
        stopAutocraft()
    
    -- !o command - Start orbiting the user who sent the command (short for orbit)
    elseif command == "!o" then
        startOrbit(speaker)
    
    -- !so command - Stop orbiting (short for stop orbit)
    elseif command == "!so" then
        stopOrbit()
    
    -- !e command - Echo text
    elseif command:sub(1,2) == "!e" then
        local text = message:sub(3):gsub("^%s+", ""):gsub("%s+$", "")
        if text ~= "" then
            sendChatMessage(text)
        else
            sendChatMessage("Usage: !e your message here")
        end
    
    -- !an command - Annoy a player
    elseif command:sub(1,3) == "!an" then
        local username = message:sub(4):gsub("^%s+", ""):gsub("%s+$", "")
        if username ~= "" then
            local targetPlayer, error = findMatchingPlayer(username, false)
            if error then
                sendChatMessage(error)
            else
                stopAnnoy()
                task.wait(0.1)
                startAnnoy(targetPlayer.Name)
            end
        else
            sendChatMessage("Usage: !an username")
        end
    
    -- !san command - Stop annoying
    elseif command == "!san" then
        stopAnnoy()
    end
end
